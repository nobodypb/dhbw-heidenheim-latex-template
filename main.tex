\input{ads/header}

\input{content/glossary}

\begin{document}

	% Cover
	\begin{spacing}{1}
		\input{ads/cover}
	\end{spacing}
  \cleardoublepage % First content should be on page 3 for double-sided printing

	\pagenumbering{Roman}
  \setcounter{page}{3} %count cover aswell
  % only page number in footer
  \pagestyle{plain}

	% Restriction notices
	\ifDocType{T2\_3100}{%
		% no restricition notices for semester paper
	}{%
		\input{ads/restrictionNotices}
    \cleardoublepage
	}%

	% Abstract
	\input{content/abstract}

	
	% space bevore chapter headline
	\RedeclareSectionCommand[beforeskip=\chapterMargin]{chapter}
  
  \newpage
	% Contents
	\begin{spacing}{1.1}
		\begingroup
		
		    % set subchapter depth
			\setcounter{tocdepth}{2}
			
			\tableofcontents
		\endgroup
	\end{spacing}
	\newpage

	% Acronyms
	\clearpage
    \input{content/acronyms}

	% List of Figures
	\clearpage
	\listoffigures

	%List of Tables
	\clearpage
	\listoftables

	% List of Listings
	\clearpage
	\lstlistoflistings
	\cleardoublepage

	\pagenumbering{arabic}
	
	\pagestyle{headings}
  \onehalfspacing

	%Content
	\foreach \i in {01,02,03,04,05,06,07,08,09,...,99} {%
		\edef\FileName{content/chapter/\i .tex}%
			\IfFileExists{\FileName}{%
				\input{\FileName}
			}
			{%
%				No chapter available
			}
	}

	\clearpage

	% Bibilography
	\cleardoublepage
	\printbibliography

	% Glossar
	\cleardoublepage
	\printglossary[style=altlist,title=\glossaryPhrase]
	\input{content/glossary}
	
	% Appendix
	\clearpage
	\appendix
	\input{content/appendix}
  
  \cleardoublepage
  % Declaration
	\input{ads/declaration}
	\newpage
	
\end{document}
